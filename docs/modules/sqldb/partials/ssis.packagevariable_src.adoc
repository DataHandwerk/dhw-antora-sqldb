CREATE View ssis.PackageVariable_src
As
Select
    p.AntoraModule
  , p.PackageName
  , T2.VariableName
  , T2.VariableDataType
  , T2.VariableValue
  , T2.ExpressionValue
  , T2.VariableHexValue
From
    ssis.Package_src       As p
    Inner Join
        ssis_t.TblVariable As T2
            On
