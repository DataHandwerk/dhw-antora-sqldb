CREATE View [ssis].[PackageParameter_tgt]
As
Select
    tgt.AntoraModule
  , tgt.PackageName
  , tgt.ParameterName
  , tgt.ParameterDataType
  , tgt.ParameterValue
  , tgt.ExpressionValue
  , tgt.ParameterHexValue
From
    ssis.PackageParameter As tgt
Where
    Exists
(
    Select
        1
    From
        ssis.AntoraModule_tgt_filter As f
    Where
        tgt.AntoraModule = f.AntoraModule
)
