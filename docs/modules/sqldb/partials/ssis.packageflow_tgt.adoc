CREATE View [ssis].[PackageFlow_tgt]
As
Select
    tgt.AntoraModule
  , tgt.PackageName
  , tgt.TaskFrom
  , tgt.Taskto
  , tgt.FlowType
  , tgt.LogicalAnd
  , tgt.ObjectName
  , tgt.CreationName
  , tgt.DTSID
  , tgt.EvalOp
  , tgt.Expression
From
    ssis.PackageFlow As tgt
Where
    Exists
(
    Select
        1
    From
        ssis.AntoraModule_tgt_filter As f
    Where
        tgt.AntoraModule = f.AntoraModule
)
