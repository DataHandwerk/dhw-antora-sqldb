CREATE View [ssis].[PackageConnection_src]
As

    p.AntoraModule
  , p.PackageName
  , T2.ConnectionManagerID
  , T2.ConnectionManagerName
  , T2.ConnectionManagerType
  , T2.ConnectionString
  , T2.ExpressionValue
  , T2.RetainSameConnectionProperty
From
    ssis.Package_src         As p
    Inner Join
        ssis_t.TblConnection As T2
            On
            p.RowID = T2.RowID
