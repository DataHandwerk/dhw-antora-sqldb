CREATE View [ssis].[PackageVariable_tgt]
As
Select
    tgt.AntoraModule
  , tgt.PackageName
  , tgt.VariableName
  , tgt.VariableDataType
  , tgt.VariableValue
  , tgt.ExpressionValue
  , tgt.VariableHexValue
From
    ssis.PackageVariable As tgt
Where
    Exists
(
    Select
        1
    From
        ssis.AntoraModule_tgt_filter As f
    Where
        tgt.AntoraModule = f.AntoraModule
)
