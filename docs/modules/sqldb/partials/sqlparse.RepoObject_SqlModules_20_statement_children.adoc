-- --a statement should have only one child
-- --if this is not the case we need to include into ROW_NUMBER()

-- , [RowNumber_per_Object] = ROW_NUMBER() OVER (
--  PARTITION BY [T1].[RepoObject_guid] ORDER BY TRY_CAST([j1].[key] AS INT)
--  )
-- --, j1.[value]
