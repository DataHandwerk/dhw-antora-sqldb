CREATE View [ssas].[IndexColumn_union]

  , index_column_id
  , RepoObjectColumn_guid
  , is_descending_key
  , is_index_primary_key
  , is_index_unique

  , tables_columns_type
  , RepoObject_guid
  , TableName

    ssas.IndexColumn_from_IsKey

    index_name           = Cast(index_name As NVarchar(450))
  , index_column_id
  , RepoObjectColumn_guid
  , is_descending_key
  , is_index_primary_key
  , is_index_unique
  , databasename
  --, TableID
  , ColumnName           = tables_columns_name
  , tables_columns_type
  , RepoObject_guid
  , TableName
From
    ssas.IndexColumn_from_IsUnique
--exlcude columns, already provided in [ssas].[IndexColumn_from_IsKey]
Where
    is_index_primary_key = 0

    index_name           = Cast(index_name As NVarchar(450))
  , index_column_id

  , is_descending_key
  , is_index_primary_key
  , is_index_unique
  , databasename

  , TableName
