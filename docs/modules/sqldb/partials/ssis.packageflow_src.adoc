CREATE View [ssis].[PackageFlow_src]

  , p.PackageName
  , T2.TaskFrom
  , T2.TaskTo
  , T2.FlowType
  , T2.LogicalAnd
  , T2.ObjectName
  , T2.CreationName
  , T2.DTSID
  , T2.EvalOp
  , T2.Expression

    ssis.Package_src                   As p
    Inner Join
        ssis_t.TblPrecedenceConstraint As T2
            On
            p.RowID = T2.RowID
